#  세 개의 양의 정수 A, B, C를 입력 받아 홀수의 개수를 리턴하는 함수를 설계 
---

**문제**

```c
#include <stdio.h>

int Solve(int A, int B, int C) 
{
	int cnt = 0;
	// 여기서부터 작성
	
	return cnt;
}

int main(void)
{
	int A, B, C;
	int sol;
	scanf("%d %d %d", &A, &B, &C);

	sol = Solve(A, B, C);

	printf("%d", sol);

	return 0;
}
```

```
입력 예시
7 5 9
```

```
출력 예시
3
```

**정답**
```c 
#include <stdio.h>

int Solve(int A, int B, int C)
{
	int cnt = 0;
	cnt = A % 2 + B % 2 + C % 2;
	return cnt;
}

int main(void)
{
	int A, B, C;
	int sol;
	scanf("%d %d %d", &A, &B, &C);

	sol = Solve(A, B, C);

	printf("%d", sol);

	return 0;
}
```
