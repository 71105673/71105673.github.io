**정답**
```c 
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

float compute_circle_area(float radious);

int main(void)
{
	float r;
	scanf("%f", &r);
	printf("%f\n", compute_circle_area(r));
}

float compute_circle_area(float radious)
{
	float pi = 3.14f;

	return radious * radious * pi;
}
```
